def check
  ::Users::CheckVerificationCode.(verification_code_params[:code], success: ->(authentication_token) {

      status:               "ok",
      authentication_token: authentication_token
    }, status: :ok

    render json: {
      status: "error",
      error:  error
    }, status: :unprocessable_entity
  })
